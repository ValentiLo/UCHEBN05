    """–ö–ª–∞—Å—Å –¥–ª—è —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è –ø—Ä–æ—Ü–µ—Å—Å–∞ —É—Å—Ç–∞–Ω–æ–≤–∫–∏"""

        print("üîç Checking system requirements...")
        
        requirements = {
            "python_version": self._check_python_version(),
            "memory": self._check_memory(),
            "disk_space": self._check_disk_space(),
            "docker": self._check_docker(),
            "docker_compose": self._check_docker_compose(),

            raise Exception("Insufficient disk space: minimum 10GB required")

    def _check_python_version(self):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –≤–µ—Ä—Å–∏–∏ Python"""
        version = sys.version_info
        return {
            "version": f"{version.major}.{version.minor}.{version.micro}",
            "meets_requirement": version >= (3, 8)
        }
    
    def _check_memory(self):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –ø–∞–º—è—Ç–∏"""
        try:
            import psutil
            memory = psutil.virtual_memory()
            return {
                "total_mb": round(memory.total / 1024 / 1024),
                "available_mb": round(memory.available / 1024 / 1024),
                "percent_used": memory.percent
            }
        except ImportError:
            return {"available_mb": 0, "error": "psutil not available"}
    
    def _check_disk_space(self):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –º–µ—Å—Ç–∞ –Ω–∞ –¥–∏—Å–∫–µ"""
        try:
            import psutil
            disk = psutil.disk_usage('/')
            return {
                "total_gb": round(disk.total / 1024 / 1024 / 1024, 1),
                "free_gb": round(disk.free / 1024 / 1024 / 1024, 1),
                "percent_used": disk.percent
            }
        except ImportError:
            return {"free_gb": 0, "error": "psutil not available"}
    
    def _check_docker(self):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ —É—Å—Ç–∞–Ω–æ–≤–∫–∏ Docker"""
        try:
            result = subprocess.run(
                ["docker", "--version"],
                capture_output=True,
                text=True,
                timeout=10
            )
            return {
                "installed": result.returncode == 0,
                "version": result.stdout.strip() if result.returncode == 0 else "Not installed"
            }
        except (subprocess.TimeoutExpired, FileNotFoundError):
            return {"installed": False, "version": "Not installed"}
    
    def _check_ports_availability(self):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–æ—Å—Ç—É–ø–Ω–æ—Å—Ç–∏ –ø–æ—Ä—Ç–æ–≤"""
        ports_to_check = [8000, 5432, 6379, 8080]
        available_ports = []
        occupied_ports = []
        
        for port in ports_to_check:
            if self._is_port_available(port):
                available_ports.append(port)
            else:
                occupied_ports.append(port)
                
        return {
            "available_ports": available_ports,
            "occupied_ports": occupied_ports
        }
    
    def _is_port_available(self, port):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–æ—Å—Ç—É–ø–Ω–æ—Å—Ç–∏ –ø–æ—Ä—Ç–∞"""
        with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
            try:
                s.bind(('localhost', port))
                return True
            except socket.error:
                return False

    def test_dependencies_installation(self):
        """–¢–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ —É—Å—Ç–∞–Ω–æ–≤–∫–∏ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–µ–π"""

Django==4.2.7
djangorestframework==3.14.0
psycopg2-binary==2.9.7
redis==4.5.5
celery==5.3.1
pillow==10.0.0
gunicorn==21.2.0
        """

            f.write(test_requirements)
        
        # –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏
        try:

                "pip", "install", "-r", "test_requirements.txt"
            ], capture_output=True, text=True, timeout=300)
            
            if result.returncode != 0:
                raise Exception(f"Dependency installation failed: {result.stderr}")
            
            # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ –≤—Å–µ –ø–∞–∫–µ—Ç—ã —É—Å—Ç–∞–Ω–æ–≤–∏–ª–∏—Å—å
            installed_packages = self._get_installed_packages()
            required_packages = ["Django", "djangorestframework", "psycopg2-binary"]
            
            missing_packages = []
            for package in required_packages:
                if package.lower() not in [pkg.lower() for pkg in installed_packages]:
                    missing_packages.append(package)
            
            if missing_packages:
                raise Exception(f"Missing packages: {missing_packages}")
            
            return {
                "installed_packages": installed_packages,
                "installation_log": result.stdout
            }
            
        finally:
            # –û—á–∏—Å—Ç–∫–∞
            if os.path.exists("test_requirements.txt"):
                os.remove("test_requirements.txt")
    
    def _get_installed_packages(self):
        """–ü–æ–ª—É—á–µ–Ω–∏–µ —Å–ø–∏—Å–∫–∞ —É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–Ω—ã—Ö –ø–∞–∫–µ—Ç–æ–≤"""
        result = subprocess.run([
            "pip", "freeze"
        ], capture_output=True, text=True)
        
        if result.returncode == 0:
            return [line.split('==')[0] for line in result.stdout.strip().split('\n')]
        return []
